import pandas as pd
import matplotlib.pyplot as plt

#     Excel c  UNIT  (   )
df_UNIT = pd.read_excel('CL-DA-Project_2-Unit_economy-team1.xlsx', sheet_name='CALC', header=3)
df_UNIT = df_UNIT[['City', 'Average Profit per Order, $.1', 'Province']]
df_UNIT = df_UNIT.rename(columns={'City': 'city', 'Province': 'province','Average Profit per Order, $.1': 'average profit per order'})

#     CSV
df_sity_info = pd.read_csv("us-cities-table.csv") #    
df = pd.read_csv('Datafiniti_Fast_Food_Restaurants_May19.csv') #   

#       
df_group = df.groupby(['city', 'province']).size().reset_index(name='count')

#      (join)
merged_df = pd.merge(df_group, df_UNIT[['city', 'province','average profit per order']], on=['city', 'province'], how='inner')
merged_df = pd.merge(merged_df, df_sity_info[['city', 'pop2024', 'aland']], on='city', how='inner')

#    (-  / 100000 )
merged_df['ratio'] = (merged_df['count'] / merged_df['pop2024']) * 100000

#     verage profit per order
sorted_df = merged_df.sort_values(by='average profit per order', ascending=False)

#  
plt.figure(figsize=(12, 6))

#   y   
ax1 = plt.gca()
ax1.bar(sorted_df['city'], sorted_df['ratio'], color='darkblue', label='The density of restaurants')
ax1.set_ylabel('number of restaurants / 100,000 people', color='blue')

#    y   
ax2 = ax1.twinx()
ax2.plot(sorted_df['city'], sorted_df['average profit per order'], color='red', marker='o', label='Average Profit per Order')
ax2.set_ylabel('average profit per order ($)', color='red')

#  
plt.title('Comparison of Cities by Number of Establishments and Average Revenue per Order')
plt.xticks(rotation=90)

#  
lines, labels = ax1.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()
ax2.legend(lines + lines2, labels + labels2, loc='upper right')

#  
plt.tight_layout()
plt.show()

import folium
from ipywidgets import interact, widgets
from geopy.geocoders import Nominatim
from IPython.display import display

#  
geolocator = Nominatim(user_agent="fast_food_locator")

#      ,       
big_chain = ['mcdonalds', 'starbucks', 'subway', 'kfc', 'burger king']

#         
def display_city_info(city):
    city_data = sorted_df[sorted_df['city'] == city]           #      
    count = city_data.iloc[0]['count']                         #   
    order = city_data.iloc[0]['average profit per order']      #     
    
    #   
    print(f"For {city}:")
    print(f"Number of restaurants: {count}")
    print(f"Average profit per order: {order}")
    
    #        ( )
    location = geolocator.geocode(city, country_codes='US')
    if location:
        latitude = location.latitude
        longitude = location.longitude
        #  
        print(f"Latitude: {latitude}, Longitude: {longitude}")
        
        #   
        city_map = folium.Map(location=[latitude, longitude], zoom_start=10)
        
        #       
        for index, row in df[df['city'] == city].iterrows():
            if row['name'].lower() in big_chain:
                #     ( )
                folium.Marker([row['latitude'], row['longitude']], popup=row['name'], icon=folium.Icon(color='red')).add_to(city_map)
            else:
                #    
                folium.Marker([row['latitude'], row['longitude']], popup=row['name'], icon=folium.Icon(color='blue')).add_to(city_map)
        
        #  
        display(city_map)
    else:
        print("Coordinates not found.")

#      
cities = sorted_df['city'].unique()

#       
city_dropdown = widgets.Dropdown(options=cities, description='Select a city:')

#   interact        
interact(display_city_info, city=city_dropdown);
